\section{Overview}

REF-Manager provides tools for:
\begin{itemize}
    \item Tracking research outputs with comprehensive metadata
    \item Three-component O/S/R quality assessments (Originality, Significance, Rigour)
    \item DOI auto-fetch for automatic metadata population
    \item Managing internal and external reviews
    \item Open Access compliance tracking
    \item REF narrative statements support
    \item Optimising submission portfolios
\end{itemize}

\section{What's New in Version 4.0}

\subsection{O/S/R Rating System}
\begin{itemize}
    \item Three-component quality ratings on 0.00-4.00 scale
    \item Ratings from Internal Panel, Critical Friend, and Self-Assessment
    \item Automatic average calculations
\end{itemize}

\subsection{DOI Auto-Fetch}
\begin{itemize}
    \item Paste DOI to auto-populate metadata
    \item Integration with OpenAlex database
    \item Fetches title, authors, venue, OA status, citations
\end{itemize}

\subsection{Enhanced Bulk Import}
\begin{itemize}
    \item Three modes: Hybrid, Smart, Manual
    \item Duplicate detection
    \item Auto-link to colleagues
\end{itemize}

\subsection{Open Access Compliance}
\begin{itemize}
    \item Track acceptance, deposit, embargo dates
    \item Automatic 3-month rule verification
    \item Exception recording
\end{itemize}

\section{Quick Installation}

\begin{enumerate}
    \item Clone repository: \texttt{git clone https://github.com/gtsoulas/ref-manager.git}
    \item Create virtual environment: \texttt{python -m venv venv}
    \item Activate: \texttt{source venv/bin/activate}
    \item Install dependencies: \texttt{pip install -r requirements.txt}
    \item Install requests: \texttt{pip install requests}
    \item Configure: \texttt{cp env.example .env}
    \item Migrate: \texttt{python manage.py migrate}
    \item Create admin: \texttt{python manage.py createsuperuser}
    \item Setup roles: \texttt{python manage.py setup\_roles --create-profiles --superusers-admin}
    \item Run: \texttt{python manage.py runserver}
\end{enumerate}
