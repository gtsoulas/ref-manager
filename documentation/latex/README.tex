\begin{tcolorbox}[colback=refblue!10, colframe=refblue]
\textbf{REF-Manager} is a comprehensive web-based application designed to help UK university departments manage their Research Excellence Framework (REF) submissions for REF 2029.
\end{tcolorbox}

\tableofcontents
\newpage

\section{Overview}

REF-Manager provides tools for:
\begin{itemize}
    \item Tracking research outputs with comprehensive metadata
    \item Coordinating quality assessments from multiple sources
    \item Managing internal and external reviews
    \item Assessing and mitigating submission risks
    \item Optimising submission portfolios
    \item Generating reports and exports
\end{itemize}

\section{Key Features}

\subsection{Research Output Management}
\begin{itemize}
    \item Multiple publication types (articles, books, chapters, etc.)
    \item Complete bibliographic metadata
    \item BibTeX and CSV import
    \item Status workflow tracking
\end{itemize}

\subsection{Multi-Dimensional Quality Assessment}
\begin{itemize}
    \item Internal Panel ratings
    \item External Critical Friend ratings
    \item Self-Assessment ratings
    \item Automatic average calculation
\end{itemize}

\subsection{Risk Assessment Framework}
\begin{itemize}
    \item Content risk scoring
    \item Timeline risk based on publication status
    \item Open Access compliance tracking
    \item Configurable risk weights
\end{itemize}

\subsection{Portfolio Optimisation}
\begin{itemize}
    \item Submission scenario modelling
    \item Quality and risk metrics
    \item Representativeness scoring
    \item Equality and diversity metrics
\end{itemize}

\subsection{Role-Based Access Control}
\begin{itemize}
    \item Four user roles: Administrator, Observer, Internal Panel, Colleague
    \item Multi-role support
    \item Granular permissions
\end{itemize}

\section{Quick Installation}

\begin{enumerate}
    \item Clone repository: \texttt{git clone https://github.com/gtsoulas/ref-manager.git}
    \item Create virtual environment: \texttt{python -m venv venv}
    \item Activate: \texttt{source venv/bin/activate}
    \item Install dependencies: \texttt{pip install -r requirements.txt}
    \item Configure: \texttt{cp env.example .env}
    \item Migrate: \texttt{python manage.py migrate}
    \item Create admin: \texttt{python manage.py createsuperuser}
    \item Setup roles: \texttt{python manage.py setup\_roles --create-profiles --superusers-admin}
    \item Run: \texttt{python manage.py runserver}
\end{enumerate}
