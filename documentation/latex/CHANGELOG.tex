\section{Version 4.0.0 -- December 2025}

\subsection*{``REF 2029 Ready'' Release}

\subsection{Added}

\subsubsection{O/S/R Rating System}
\begin{itemize}
    \item Three-component decimal ratings (0.00-4.00) for Originality, Significance, Rigour
    \item Ratings from three sources: Self-Assessment, Internal Panel, Critical Friend
    \item Nine new model fields for O/S/R ratings
    \item Computed average properties
\end{itemize}

\subsubsection{DOI Auto-Fetch}
\begin{itemize}
    \item OpenAlex API integration for automatic metadata population
    \item Auto-populates: title, authors, venue, year, OA status, citations
    \item Endpoint: \texttt{/outputs/fetch-doi/}
\end{itemize}

\subsubsection{Enhanced Bulk Import}
\begin{itemize}
    \item Three import modes: Hybrid, Smart, Manual
    \item Duplicate detection by DOI and title
    \item Auto-link to colleagues via staff\_id
\end{itemize}

\subsubsection{Open Access Compliance}
\begin{itemize}
    \item New fields: acceptance\_date, deposit\_date, embargo\_end\_date
    \item OA status tracking (Gold, Green, Hybrid, Bronze, Closed)
    \item Automatic 3-month deposit rule compliance checking
\end{itemize}

\subsubsection{REF Narrative Statements}
\begin{itemize}
    \item Double-weighting statement field (300 word limit)
    \item Interdisciplinary statement field (500 word limit)
    \item Real-time word counting
\end{itemize}

\subsection{Changed}
\begin{itemize}
    \item Output edit form redesigned with card-based layout
    \item Removed legacy star-rating dropdown fields
    \item Updated admin configuration
\end{itemize}

\subsection{Fixed}
\begin{itemize}
    \item admin.py: Fixed field references
    \item forms.py: Removed auto\_now\_add conflict
    \item views.py: Fixed related\_name issues
    \item views.py: Fixed OpenAlex API URL format
\end{itemize}

\section{Upgrade Notes}

\subsection{Upgrading to 4.0 from 3.x}

\begin{enumerate}
    \item Backup your database
    \item Install requests: \texttt{pip install requests}
    \item Run migrations: \texttt{python manage.py migrate}
    \item Replace updated files
    \item Collect static: \texttt{python manage.py collectstatic}
    \item Restart services
\end{enumerate}
